{	uchar number1,number2,number3;

    Initial_Dispay_Line(0x00);

#pragma disable 
void Write_Data(unsigned char dat)
{unsigned char i; 
	CS1=0;RS=1;
	for(i=0;i<8;i++) 
	 { 
		SDA=(bit)(dat&0x80); 
		SCLK=0; 
			_nop_();
		SCLK=1; 
		dat=dat<<1; 
	 }

	CS1=1;
	return;
}



#pragma disable
void Write_Instruction(unsigned char cmd)
{unsigned char i; 
	CS1=0;RS=0;
	for(i=0;i<8;i++) 
	 { 
		SDA=(bit)(cmd&0x80); 
		SCLK=0; 
			_nop_();
		SCLK=1; 
		cmd=cmd<<1; 
	 }

	CS1=1;
	return;
}






//Specify DDRAM line for COM0 0~63
void Initial_Dispay_Line(unsigned char line)
{
    line=line|0x40;
    Write_Instruction(line);
	return;
}



void Set_Page_Address(unsigned char add)
{
    add=0xb0|add;
    Write_Instruction(add);
	return;
}

void Set_Column_Address(unsigned char add)
{
    Write_Instruction((0x10|(add>>4)));
	Write_Instruction((0x0f&add));
	return;
}





void Set_Contrast_Control_Register(unsigned char mod)
{
    Write_Instruction(0x81);
	Write_Instruction(mod);
	return;
}




#pragma disable
void Initial(void)
{

	RST=1;
	Delay(200);
	RST=0;
	Delay(2000);
	RST=1;
	Delay(1000);
	CS1=0;

  	Write_Instruction(0xe2);//system reset

  	Write_Instruction(0xa0);//set Frame Rate[A0: 76fps, A1b: 95fps, A2b: 132fps, A3b: 168fps(fps: frame-per-second)] 
 
  	Write_Instruction(0xeb);//set BR 
  	Write_Instruction(0x2f);//set Power Control

  	Write_Instruction(0xc4);//set LCD Mapping Control
  	Write_Instruction(0x81);//set PM
  	Write_Instruction(Contrast_level); //set PM 110
  	Write_Instruction(0xaf);//display Enable
}



void Display_Chinese(unsigned char ft[])
{
    unsigned char i,j,k,b,c=0;
	unsigned int	num=0;

for(b=0;b<4;b++)
  {
    for(i=0;i<0x02;i++)
	{	Set_Page_Address(c);
    	Set_Column_Address(0x00);
	    num=i*0x10+b*256;
		for(j=0;j<12;j++)
		{
            for(k=0;k<16;k++)
			{
		        Write_Data(ft[num+k]);
			}
			num+=0x20;
		}c++;
	}
  }
    return;
}
